/*function to enable pausing within an async function using the await keyword*/
function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

/* ########################################
   Function manipulates the "rendered" page
   ######################################## */
jQuery(document).ready(function () {
  console.log("Window Loaded");

  if (flag) {
    /* Copy "Move How did I Do?" button to response areas*/
    // console.log("Attemting to move How Did I Do Button on First Tab");
    /* Removing doInPreview or doInSession removes mobius' margin rule */
    jQuery(".btn.doInSession, .btn.doInPreview")
      .detach()
      .insertAfter(jQuery(".parts-container .part")
        .map((ind, cur) => {
          let resWrapper = jQuery(cur).find(".response-wrapper").last();
          if (resWrapper.find(".check-help-container").length) {
            return resWrapper.find(".check-help-container").prev();
          } else {
            return resWrapper.children().last();
          }
        }))
      .addClass("check-help-button")
      .text("Check");

    /* Select all response areas in the parts that are not active */
    var inactiveFields = jQuery(".parts-container .part:not(.active) .response-wrapper .response, #mark-as-done-container .response");

    /* Soft hide them so that they're not taken into account when "how did I do?"
    This is done by preceding the input's id with "hidden-" so it isn't seen during the check*/
    inactiveFields.each(function () {
      jQuery(this).attr("data-original-id", jQuery(this).attr("id"));
      jQuery(this).attr("id", "hidden-" + jQuery(this).attr("id"));
    });

    /* Still need to add original-id attr to the active part's response areas */
    var activeFields = jQuery(".parts-container .part.active .response-wrapper .response");
    activeFields.each(function () {
      jQuery(this).attr("data-original-id", jQuery(this).attr("id"));
    });

    /* NAV BAR EDITS: */
    /* Remove mention of "Page" in navigation bar links */
    jQuery("#assignmentButtons .container .btn").each(function () {
      jQuery(this).text(jQuery(this).text().replace("Page", ""));
    });

    /* Select quit and save button, and move to "safe ul" */
    jQuery("#assignmentButtons .container a[id='MenuItem.actions.assignment.quit_save']").prependTo("#assignmentButtons .container ul:eq(1)");

    /* Select quit and save button, and move to "safe ul" */
    jQuery("#assignmentButtons .container a[id='MenuItem.actions.lesson.grade']").prependTo("#assignmentButtons .container ul:eq(1)").text("Submit");

    /* Delete first ul (contains useless buttons) */
    jQuery("#assignmentButtons .container ul").first().remove();

    /* PAGE STYLE CHANGES */
    /* Remove hover border from question container */
    jQuery("div.questionWrapper").removeClass("questionWrapper");

    /* Remove questionButtons */
    jQuery("div.questionButtons").css("display", "none");

    /* Remove Number/Unit help buttons */
    jQuery(".helpLink").css("display", "none");

    /* COMMENT THREAD */
    /* Insert Comment thread code after the question form generated by Mobius */
    async function installComments() {
      const commentThreadSnippet = await (await fetch("/web/Pjohnso000/Public_Html/Scripts/CommentThread.html")).text();
      // console.log("Fetched thread HTML from Mobius server");
      jQuery(commentThreadSnippet).insertAfter("form[name='edu_form']");

      while (!window["comments"]) {
        /*allow event loop to run other scripts while waiting for comments to exist*/
        await sleep(100);
      }
      console.log("Comments are live!");

      showNumberOfComments();
      formatComments();
      formatNameField();
      modSubmitBtn_Different();
      modHcbIcons();
    }

    installComments();
  }

  flag = false;
});

/* ###############################
   Function handles tab behaviours
   ############################### */
jQuery(".tabs-container .tab").on("click", function () {
  /* Use regex to find selected tab's number */
  let number = jQuery(this).attr("id").match(/\d+/)[0];
  // console.log("Clicked on tab number " + number);

  if (!jQuery(this).hasClass("active")) {
    /* Disable response areas from previously active part */
    jQuery(".parts-container .part.active .response-wrapper .response").each(function () {
      jQuery(this).attr("id", "hidden-" + jQuery(this).attr("data-original-id"));
      /* Need to trigger the click event handler as that's how we can make the "how did I do?" button to come back */
      jQuery(this).trigger("click");
    });

    /* Remove active class from previously active part and tab */
    jQuery(".parts-container .part.active").toggleClass("active");
    jQuery(".tabs-container .tab.active").toggleClass("active");

    /* Toggle active class on tab and corresponding part */
    jQuery(this).toggleClass("active");
    var activePart = jQuery(".parts-container #part" + number);
    activePart.toggleClass("active");

    /* Move "How did I Do?" button to the active part's container */
    // console.log("Attemting to move How Did I Do Button");
    // jQuery(".btn.doInSession, .btn.doInPreview").appendTo(".parts-container .part.active .response-wrapper").text("🌊 How Did I Do? 🌊");

    /* Un-hide response areas in the newly active part */
    activePart.find(".response-wrapper .response").each(function () {
      jQuery(this).attr("id", jQuery(this).attr("data-original-id"));
    });
  }
});

/* #########################################################
   Functions handle Step by Step Module Navigation
   ######################################################### */
/* Step down logic */
jQuery(".sbs-step-down").on("click", function () {
  var navContainer = jQuery(this).parent();
  var nextEl = navContainer.next();
  nextEl.toggleClass("active").insertBefore(navContainer);
  if (nextEl.hasClass("last")) {
    /* Disable buttons */
    jQuery(this).addClass("disabled");
    navContainer.children(".sbs-reveal-all").addClass("disabled");
  }
  /* If this button is pressed, we can always use the other one */
  navContainer.children(".sbs-step-up").removeClass("disabled");
});
/* Step up logic */
jQuery(".sbs-step-up").on("click", function () {
  var navContainer = jQuery(this).parent();
  var prevEl = navContainer.prev();
  prevEl.toggleClass("active").insertAfter(navContainer);
  if (prevEl.hasClass("second")) {
    /* Disable the button */
    jQuery(this).addClass("disabled");
  }
  /* If this button is pressed, we can always use the other one */
  navContainer.children(".sbs-step-down, .sbs-reveal-all").removeClass("disabled");
});
/* Reveal all logic */
jQuery(".sbs-reveal-all").on("click", function () {
  var navContainer = jQuery(this).parent();
  /* Insert nav bar after last worked soln part */
  navContainer.insertAfter(navContainer.parent().children(".last"));
  /* Allow all parts to be active */
  navContainer.parent().children(".sbs-step").addClass("active");
  /* Update state of other nav buttons */
  navContainer.children(".sbs-step-up").removeClass("disabled");
  navContainer.children(".sbs-step-down").addClass("disabled");
  jQuery(this).addClass("disabled");
});
